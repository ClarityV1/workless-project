document.getElementById('start').addEventListener('click', () => { // 1. Find the tab you are currently looking at chrome.tabs.query({active: true, currentWindow: true}, (tabs) => { const activeTab = tabs[0]; // 2. Send the "START" message to that tab chrome.tabs.sendMessage(activeTab.id, {action: "START_RECORDING"}, (response) => { if (chrome.runtime.lastError) { // If the page hasn't been refreshed, it might not "hear" the extension document.getElementById('status').innerText = "❌ Refresh the page first!"; } else { document.getElementById('status').innerText = "🔴 Learning..."; } }); }); }); // We keep the Stop button logic too document.getElementById('stop').addEventListener('click', () => { chrome.tabs.query({active: true, currentWindow: true}, (tabs) => { chrome.tabs.sendMessage(tabs[0].id, {action: "STOP_RECORDING"}, (response) => { if (response && response.data) { document.getElementById('status').innerText = "✅ Task Saved!"; console.log("Captured Data:", response.data); } }); }); });
